 72         function getFrame(id) {
 73         	// We need to get the index of the iframe in case there
 74         	// are multiple iframes on the page.
 75         	// Referencing the iframe via frames['canvas-frame'+id] is
 76         	// no good because the location is clobbered if you reference
 77         	// it in that way (but only the second time you reference it - weird),
 78         	// which causes postMessage to break.
 79 
 80             var i, length, ifs, fn = "canvas-frame-" + id;
 81             ifs = window.document.getElementsByTagName('iframe');
 82             length = ifs.length;
 83             for (i = 0; i < length; i += 1) {
 84                 try {
 85                     if (ifs[i].name === fn){
 86                         return ifs[i].contentWindow;
 87                     }
 88                 }
 89                 catch (e) {}
 90             }
 91             // This is bad, and is only used as a fallback and will not work second time the
 92             // app is clicked on
 93             return frames[fn];
 94         }

109         function xdCallback(data, ctx) {
110 
111             var id = ctx.id;
112 
113             if ($$.isNil(id)) {
114                 throw getLabel("CanvasParentJs", "missingIdInContext");
115             }
116 
117             if (data) {
118 
119                 var type = data.body.type,
120                     seq  = data.seq,
121                     can  = canvases[id];
122 
123                 // Update the version of the client.
124                 can.cversion = data.clientVersion;
125 
126                 if (type === 'ctx') {
127                     submodules.services(seq, id).ctx(data, can);
128                 }
129                 else if (type === 'ajax') {
130                     submodules.services(seq, id).ajax(data);
131                 }
132                 else if (type === 'resize') {
133                     submodules.frame.resize(data, can);
134                 }
135                 else if (type === 'subscribe') {
136                     submodules.events.subscribe(data, can);
137                 }
138                 else if (type === 'unsubscribe') {
139                     submodules.events.unsubscribe(data, can);
140                 }
141                 else if (type === 'publish') {
142                     submodules.events.publish(data, can);
143                 }
144             }

179                 function subscribeOnScroll() {
180 
181                     // If already subscribing to the window.onscroll event.
182                     if (onscroll === true) {
183                         return;
184                     }
185                     onscroll = true;
186 
187                     // Add  event while preserving any existing listeners
188                     if (window.addEventListener) {
189                         window.addEventListener('scroll', onScroll, false);
190                     }
191                     else if (window.attachEvent) {
192                         window.attachEvent('onscroll', onScroll);
193                     }
194                 }

196                 function unSubscribeOnScroll() {
197 
198                     var found = false;
199 
200                     if (onscroll === false) {
201                         return;
202                     }
203 
204                     // If any of the handlers are still listening don't stop
205                     $$.each(canvases, function (c) {
206                         if (!$$.isNil(c.handlers.scroll)) {
207                             found = true;
208                         }
209                     });
210 
211                     if (! found) {
212 
213                         if (window.removeEventListener) {
214                             window.removeEventListener('scroll', onScroll, false);
215                         }
216                         else if (window.detachEvent) {
217                             window.detachEvent('onscroll', onScroll);
218                         }
219                         onscroll = false;
220                     }
221                 }

224                     subscribe : function (data, canvas) {
225                         var subscriptions = data.body.subscriptions;
226                         canvas.handlers = subscriptions;
227

229                         if (!$$.isNil(subscriptions.scroll)) {
230                             subscribeOnScroll();
231                         }
232                         if (!$$.isNil(subscriptions.streaming)) {
233                             // console.log(This event reserved for streaming API", subscriptions.streaming.params);
234                         }
235                     },

248                             unSubscribeOnScroll();
249                         }
250                         if (!$$.isNil(subscriptions.streaming)) {
251                             // Handle any special case for streaming.
252                         }
253 
254                     },
255                     publish : function (data) {

532             var frameDoc, frameObj, frameElem;

534             // Remove all child elements from the last time we rendered a canvas app
535             while ( parent.firstChild ) {parent.removeChild( parent.firstChild );}
536 
537             frameElem =  parent.appendChild(buildIFrame(id,uid,options));
538             canvases[id].frameElement = frameElem;
539             frameObj = parent.appendChild(frameElem);
540             if (frameObj.contentDocument) {
541                 frameDoc = frameObj.contentDocument;
542             }
543             else if (frameObj.contentWindow) {
544                 frameDoc = frameObj.contentWindow.document;
545             }
546             frameDoc.location.replace(canvases[id].url);

549         /*
550          * Decodes and parses the signed request.
551          */
552         function parseMetadataFromSignedRequest(id, signedRequest){
553         	var parts = signedRequest.split('.');
554         	
555         	if (parts.length !== 2){
556         		throw (getLabel("CanvasParentJs", "invalidSignedRequestFormat"));
557         	}
558         	
559         	var srObject = Sfdc.JSON.parse(Sfdc.String.base64Decode(parts[1]));
560         	if ($$.isNil(srObject) || $$.isNil(srObject.context.application)){
561         		throw (getLabel("CanvasParentJs", "unableToParseApp"));
562         	}
563         	canvases[id].app = srObject.context.application;
564         	canvases[id].loginUrl = srObject.context.links.loginUrl;
565         }

567         /*
568          * Responsible for parsing/decoding the signed request and rendering.  
569          */
570         function processSignedRequestAndRender(id,containerElem,uid,options){
571             parseMetadataFromSignedRequest(id,canvases[id].sr);
572             
573             var curl = decodeURIComponent(canvases[id].app.canvasUrl), 
574             	parentUrl = document.location.protocol + "//" + document.location.host;
575 
576             if (canvases[id].app.authType === "OAUTH") {
577             	var q;
578                 if (! $$.isNil(canvases[id].loginUrl) && "https://login.salesforce.com" !== canvases[id].loginUrl) {
579                     q = $$.param({loginUrl : encodeURIComponent(canvases[id].loginUrl)});
580                 }
581                 q += "#target_origin=" + encodeURIComponent(parentUrl) + "&instance_id=" + id ;
582                 canvases[id].url = $$.query(curl, q);
583                 addFrame(id, containerElem, uid, options);
584             }
585             else {
586             	canvases[id].url = curl;
587                 addSignedRequestFrame(id, canvases[id].sr, containerElem, uid, options);
588             }
589         	
590         }

592         /*
593          * Responsible for handling any errors that come back from the signed
594          * request call.  
595          */
596         function processSignedRequestErrors(id,containerElem,data){
597         	var namespace = canvases[id].app.namespace,
598         	    dataObj = Sfdc.JSON.parse(data),
599         		nameToShow = $$.isNil(canvases[id].app.name)?canvases[id].app.developerName:canvases[id].app.name;
600         	
601         	if (!$$.isNil(dataObj) && dataObj.length >= 1){
602         		// There may be multiple errors, so grab just the first one.
603         		var error = dataObj[0];
604         		switch (error.errorCode) {
605 	        	    case "APP_NOT_FOUND":
606 	        	    case "NO_ACCESS":
607 	        	    case "UNSUPPORTED_LOCATION":
608 	                	renderError(id,containerElem,
609 	                			nameToShow,
610 	                			error.message);
611 	                	break;
612 	        	    case "INVALID_SESSION_ID":
613 	        	    	renderError(id,containerElem,
614 	                			nameToShow,
615 	                			getLabel("CanvasParentJs","sessionInvalid"));
616 	        	    	break;
617 	        	    default:
618 	                	renderError(id,containerElem,
619 	                			nameToShow,
620 	                			getLabel("CanvasParentJs","signedRequestRetrievalError",error.errorCode + ":" + error.message));
621         		}
622         	}
623         	else {
624             	renderError(id,containerElem,
625             			nameToShow,
626             			getLabel("CanvasParentJs","signedRequestRetrievalError",""));
627         	}
628         }

630         /*
631          * Responsible for getting the signed request and rendering.  Also
632          * error handler for signed request call.
633          */
634         function renderCanvas(id,containerElem,uid,options){
635         	// Signed request was not provided from caller, so make the call to get it.
636         	if ($$.isNil(canvases[id].sr)){
637                 getSignedRequest(id,containerElem,uid,options,
638             		// Success handler
639     	            function(data) {
640                 	canvases[id].sr = Sfdc.JSON.parse(data).response;
641     	                processSignedRequestAndRender(id,containerElem,uid,options);
642     	            },
643     	            // Failure handler
644     	            function(data) {
645         	            processSignedRequestErrors(id,containerElem,data);
646     	            }
647                 );
648         	}
649         	// Signed request was provided by calling page.
650         	else{
651         		processSignedRequestAndRender(id,containerElem,uid,options);
652         	}
653         }

656         function render(container, elem, options) {
657             var optionsDefaults = {
658                 frameborder: '0',
659                 width: {value : '800px', max : "1000px"},
660                 height: {value : '900px', max : "2000px"},
661                 scrolling: 'no',
662                 displayLocation: 'NONE'
663             };
664 
665             if (options) {
666                 // Convert any old style width and height to new
667                 if (!$$.isObject(options.width) && !$$.isNil(options.width)) {
668                     options.width = $$.extend(optionsDefaults.width, {value : options.width});
