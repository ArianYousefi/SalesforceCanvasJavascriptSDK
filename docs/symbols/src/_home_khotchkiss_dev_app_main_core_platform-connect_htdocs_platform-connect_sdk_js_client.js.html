 15         function startsWithHttp(u, d) {
 16             return  $$.isNil(u) ? u : (u.substring(0, 4) === "http") ? u : d;
 17         }

 19         // returns the url of the Parent Window
 20         function getTargetOrigin(to) {
 21             var h;
 22             if (to === "*") {return to;}
 23             if (!$$.isNil(to)) {
 24                 h = $$.stripUrl(to);
 25                 purl = startsWithHttp(h, purl);
 26                 if (purl) {return purl;}
 27             }
 28             // This relies on the parent passing it in. This may not be there as the client can do a redirect.
 29             h = document.location.hash;
 30             if (h) {
 31                 h = decodeURIComponent(h.replace(/^#/, ''));
 32                 purl = startsWithHttp(h, purl);
 33             }
 34             return purl;
 35         }

 38         function xdCallback(data) {
 39             if (data) {
 40                 if (submodules[data.type]) {
 41                     submodules[data.type].callback(data);
 42                 }
 43                 else {
 44                     throw "Undefined event type " + data.type;
 45                 }
 46             }
 47         }

 55             function postit(clientscb, message) {
 56                 var wrapped, to, c;
 57 
 58                 // need to keep a mapping from request to callback, otherwise
 59                 // wrong callbacks get called. Unfortunately, this is the only
 60                 // way to handle this as postMessage acts more like topic/queue.
 61                 // limit the sequencers to 100 avoid out of memory errors
 62                 seq = (seq > 100) ? 0 : seq + 1;
 63                 cbs[seq] = clientscb;
 64                 wrapped = {seq : seq, src : "client", clientVersion : cversion, parentVersion: pversion, body : message};
 65 
 66                 c  = message && message.config && message.config.client;
 67                 to = getTargetOrigin($$.isNil(c) ? null : c.targetOrigin);
 68                 if ($$.isNil(to)) {
 69                     throw "ERROR: targetOrigin was not supplied and was not found on the hash tag, this can result from a redirect or link to another page. " +
 70                         "Try setting the targetOrgin (example: targetOrigin : sr.context.environment.targetOrigin) " +
 71                         "when making an ajax request.";
 72                 }
 73                 $$.xd.post(wrapped, to, parent);
 74             }

 76             function validateClient(client, cb) {
 77                 var msg;
 78                 if ($$.isNil(client) || $$.isNil(client.oauthToken)) {
 79                     msg = {status : 401, statusText : "Unauthorized" , parentVersion : pversion, payload : "client or client.oauthToken not supplied"};
 80                 }
 81                 if ($$.isNil(client.instanceId) || $$.isNil(client.targetOrigin)) {
 82                     msg = {status : 400, statusText : "Bad Request" , parentVersion : pversion, payload : "client.instanceId or client.targetOrigin not supplied"};
 83                 }
 84                 if (!$$.isNil(msg)) {
 85                     if ($$.isFunction(cb)) {
 86                         cb(msg);
 87                         return false;
 88                     }
 89                     else {
 90                         throw msg;
 91                     }
 92                 }
 93                 return true;
 94             }

 98             var event = (function() {
 99                 var subscriptions = {};
100                 return  {
101                     callback : function (data) {
102                         var event = data.payload,
103                             subscription = subscriptions[event.name];
104                         if (!$$.isNil(subscription)) {
105                             if ($$.isFunction(subscription.onData)) {

110                     /**
111                      * @description Subscribe to parent events. example "scroll".
112                      * @public
113                      * @name Sfdc.canvas.client#subscribe
114                      * @function
115                      * @param {client} client object from signed request.
116                      * @param {Object} s subscriber object with callback functions.
117                      * @example
118                      *
119                      * Sfdc.canvas(function() {
120                      *     sr = JSON.parse('<%=signedRequestJson%>');
121                      *     // Capture the onScrolling event of the parent window.
122                      *     Sfdc.canvas.client.subscribe(sr.client, {onScroll : function () {
123                      *          console.log("Parent's contentHeight; " + event.heights.contentHeight);
124                      *          console.log("Parent's pageHeight; " + event.heights.pageHeight);
125                      *          console.log("Parent's scrollTop; " + event.heights.scrollTop);
126                      *          console.log("Parent's contentWidth; " + event.widths.contentWidth);
127                      *          console.log(Parent's "pageWidth; " + event.widths.pageWidth);
128                      *          console.log("Parent's scrollLeft; " + event.widths.scrollLeft);
129                      *      }});
130                      * });
131                      */
132                     subscribe : function(client, s) {
133                         var subs = {};
134 
135                         if ($$.isNil(s) || !validateClient(client || $$.oauth.client())) {
136                             return;
137                         }
138 
139                         $$.each($$.isArray(s) ? s : [s], function (v) {
140                             if (!$$.isNil(v.name)) {
141                                 subscriptions[v.name] = v;
142                                 subs[v.name] = {
143                                     params : v.params
144                                 };
145                             }
146                             else {
147                                 throw "subscription does not have a 'name'";
148                             }
149                         });
150                         postit(null, {type : "subscribe", config : {client : client}, subscriptions : subs});
151                     },
152                     unsubscribe : function(client, s) {
153                         // client can pass in the handler object or just the name
154                         var subs = {};
155 
156                         if ($$.isNil(s) || !validateClient(client || $$.oauth.client())) {
157                             throw "PRECONDITION FAIL: need fo supply client and event name";
158                         }
159 
160                         if ($$.isString(s)) {
161                             subs[s] = {};
162                             delete subscriptions[s];
163                         }
164                         else {
165                             $$.each($$.isArray(s) ? s : [s], function (v) {
166                                 subs[v.name] = {
167                                     params : v.params
168                                 };
169                                 delete subscriptions[v.name];
170                             });
171                         }
172                         postit(null, {type : "unsubscribe", config : {client : client}, subscriptions : subs});
173                     },
174                     publish : function(client, e) {
175                         if (!$$.isNil(e) && !$$.isNil(e.name)) {
176                             client = client || $$.oauth.client();
177                             if (validateClient(client)) {
178                                 postit(null, {type : "publish", config : {client : client}, event : e});
179                             }
180                         }
181                     }
182                 };
183             }());

185             var callback = (function() {
186                 return  {
187                     callback : function (data) {
188                         // If the server is telling us the access_token is invalid, wipe it clean.
189                         if (data.status === 401 &&
190                             $$.isArray(data.payload) &&
191                             data.payload[0].errorCode &&
192                             data.payload[0].errorCode === "INVALID_SESSION_ID") {
193                             // Session has expired logout.
194                             $$.oauth.logout();
195                         }
196                         // Find appropriate client callback an invoke it.
197                         if ($$.isFunction(cbs[data.seq])) {
198                             if (!$$.isFunction(cbs[data.seq])) {
199                                 alert("not function");
200                             }
201                             cbs[data.seq](data);
202                         }
203                         else {
204                             // This can happen when the user switches out canvas apps real quick,
205                             // before the request from the last canvas app have finish processing.
206                             // We will ignore any of these results as the canvas app is no longer active to
207                             // respond to the results.
208                         }
209                     }
210                 };

213             var services = (function() {
214                 return  {
215                     /**
216                      * @description Perform a cross-domain, asynchronous HTTP request.
217                      <br>Note:  this should not be used for same domain requests.
218                      * @param {String} url The URL to which the request is sent
219                      * @param {Object} settings A set of key/value pairs to configure the request.
220                      <br>The success setting is required at minimum and should be a callback function
221                      * @config {String} [client] required client context {oauthToken: "", targetOrigin : "", instanceId : ""}
222                      * @config {String} [contentType] "application/json"
223                      * @config {String} [data] request body
224                      * @config {String} [headers] request headers
225                      * @config {String} [method="GET"] The type of Ajax Request to make
226                      * @config {Function} [success] Callback for all responses from server (failure and success) . Signature: success(response); intersting fields: [response.data, response.responseHeaders, response.status, response.statusText}
227                      * @config {Boolean} [async=true] Asyncronous: true is only supported at this time.
228                      * @name Sfdc.canvas.client#ajax
229                      * @function
230                      * @throws error if the URL is missing or the settings object does not contain a success callback function.
231                      * @example
232                      * //Posting To a Chatter Feed:
233                      * var sr = JSON.parse('<%=signedRequestJson%>');
234                      * var url = sr.context.links.chatterFeedsUrl+"/news/"
235                      *                                   +sr.context.user.userId+"/feed-items";
236                      * var body = {body : {messageSegments : [{type: "Text", text: "Some Chatter Post"}]}};
237                      * connect.client.ajax(url,
238                      *   {client : sr.client,
239                      *     method: 'POST',
240                      *     contentType: "application/json",
241                      *     data: JSON.stringify(body),
242                      *     success : function(data) {
243                      *     if (201 === data.status) {
244                      *          alert("Success"
245                      *          }
246                      *     }
247                      *   });
248                      * @example
249                      * // Gets a List of Chatter Users:
250                      * // Paste the signed request string into a JavaScript object for easy access.
251                      * var sr = JSON.parse('<%=signedRequestJson%>');
252                      * // Reference the Chatter user's URL from Context.Links object.
253                      * var chatterUsersUrl = sr.context.links.chatterUsersUrl;
254                      *
255                      * // Make an XHR call back to salesforce through the supplied browser proxy.
256                      * connect.client.ajax(chatterUsersUrl,
257                      *   {client : sr.client,
258                      *   success : function(data){
259                      *   // Make sure the status code is OK.
260                      *   if (data.status === 200) {
261                      *     // Alert with how many Chatter users were returned.
262                      *     alert("Got back "  + data.payload.users.length +
263                      *     " users"); // Returned 2 users
264                      *    }
265                      * })};
266                      */
267                     ajax : function (url, settings) {
268 
269                         var ccb, config, defaults;
270 
271                         if (!url) {
272                             throw "PRECONDITION ERROR: url required with AJAX call";
273                         }
274                         if (!settings || !$$.isFunction(settings.success)) {
275                             throw "PRECONDITION ERROR: function: 'settings.success' missing.";
276                         }
277                         if (! validateClient(settings.client, settings.success)) {
278                             return;
279                         }
280 
281                         ccb = settings.success;
282                         defaults = {
283                             method: 'GET',
284                             async: true,
285                             contentType: "application/json",
286                             headers: {"Authorization" : "OAuth "  + settings.client.oauthToken,
287                                 "Accept" : "application/json"},
288                             data: null
289                         };
290                         config = $$.extend(defaults, settings || {});
291 
292                         // Remove any listeners as functions cannot get marshaled.
293                         config.success = undefined;
294                         config.failure = undefined;
295                         // Don't allow the client to set "*" as the target origin.
296                         if (config.client.targetOrigin === "*") {
297                             config.client.targetOrigin = null;
298                         }
299                         else {
300                             // We need to set this here so we can validate the origin when we receive the call back
301                             purl = startsWithHttp(config.targetOrigin, purl);
302                         }
303                         postit(ccb, {type : "ajax", url : url, config : config});
304                     },
305                     /**
306                      * @description Get the context for the current user and organization
307                      * @public
308                      * @name Sfdc.canvas.client#ctx
309                      * @function
310                      * @param {Function} clientscb Callback function to run when the call to ctx is complete
311                      * @param {Object} client signedRequest.client.
312                      * @example
313                      * // Gets context in the canvas app.
314                      *
315                      * function callback(msg) {
316                      *   if (msg.status !== 200) {
317                      *     alert("Error: " + msg.status);
318                      *     return;
319                      *   }
320                      *   alert("Payload: ", msg.payload);
321                      * }
322                      * var ctxlink = connect.byId("ctxlink");
323                      * var oauthtoken = connect.oauth.token();
324                      * ctxlink.onclick=function() {
325                      *   connect.client.ctx(callback, oauthtoken)};
326                      * }
327                      */
328                     ctx : function (clientscb, client) {
329                         client = client || $$.oauth.client();
330                         if (validateClient(client, clientscb)) {
331                             postit(clientscb, {type : "ctx", accessToken : client.oauthToken, config : {client : client}});
332                         }
333                     },
334                     /**
335                      * @description Stores or gets the oauth token in a local javascript variable. Note, if longer term
336                      * (survive page refresh) storage is needed store the oauth token on the server side.
337                      * @param {String} t oauth token, if supplied it will be stored in a volatile local JS variable.
338                      * @returns {Object} the oauth token.
339                      */
340                     token : function(t) {
341                         return $$.oauth.token(t);
342                     },
343                     /**
344                      * @description Returns the current version of the client.
345                      * @returns {Object} {clientVersion : "28.0", paranetVersion : "28.0"}.
346                      */
347                     version : function() {
348                         return {clientVersion: cversion, parentVersion : pversion};
349                     }
350                 };

353             var frame = (function() {
354                 return  {
355                     /**
356                      * @public
357                      * @name Sfdc.canvas.client#size
358                      * @function
359                      * @description Convenience method for getting the current size of the iframe.
360                      * @return {Object}
361                      * heights.contentHeight - the height of the virtual iframe, all content not just visible.
362                      * heights.pageHeight - the height of the visible iframe in the browser.
363                      * heights.scrollTop - the position of the scroll bar measured from the top.
364                      * widths.contentWidth - the width of the virtual iframe, all content not just visible.
365                      * widths.pageWidth - the width of the visible iframe in the browser.
366                      * widths.scrollLeft - the position of the scroll bar measured from the left.
367                      * @example
368                      *
369                      * var sizes = Sfdc.canvas.client.size();
370                      * console.log("contentHeight; " + sizes.heights.contentHeight);
371                      * console.log("pageHeight; " + sizes.heights.pageHeight);
372                      * console.log("scrollTop; " + sizes.heights.scrollTop);
373                      * console.log("contentWidth; " + sizes.widths.contentWidth);
374                      * console.log("pageWidth; " + sizes.widths.pageWidth);
375                      * console.log("scrollLeft; " + sizes.widths.scrollLeft);
376                      *
377                      * Sfdc.canvas(function() {
378                      *     sr = JSON.parse('<%=signedRequestJson%>');
379                      *     Sfdc.canvas.client.autogrow(sr.client);
380                      * });
381                      */
382                     size : function() {
383                         var contentHeight = document.documentElement.scrollHeight,
384                             pageHeight = document.documentElement.clientHeight,
385                             scrollTop = (document.documentElement && document.documentElement.scrollTop) || document.body.scrollTop,
386                             contentWidth = document.documentElement.scrollWidth,
387                             pageWidth = document.documentElement.clientWidth,
388                             scrollLeft = (document.documentElement && document.documentElement.scrollLeft) || document.body.scrollLeft;
389 
390                         return {heights : {contentHeight : contentHeight, pageHeight : pageHeight, scrollTop : scrollTop},
391                             widths : {contentWidth : contentWidth, pageWidth : pageWidth, scrollLeft : scrollLeft}};
392                     },
393                     /**
394                      * @public
395                      * @name Sfdc.canvas.client#resize
396                      * @function
397                      * @description Informs the parent window to resize your canvas iframe. If parameters are not specified
398                      * the parent window will attempt to automatically determine the height of the canvas app based off of
399                      * content and set the iframe's width and height accordingly. If you would like to set the dimensions
400                      * explicitly pass in an object with height and/or width properties.
401                      * @param {Client} client object from signed request.
402                      * @param {size} size optional height and width information
403                      * @example
404                      * //Automatically determine the size
405                      * Sfdc.canvas(function() {
406                      *     sr = JSON.parse('<%=signedRequestJson%>');
407                      *     Sfdc.canvas.client.resize(sr.client);
408                      * });
409                      *
410                      * @example
411                      * //Set the height and width explicitly
412                      * Sfdc.canvas(function() {
413                      *     sr = JSON.parse('<%=signedRequestJson%>');
414                      *     Sfdc.canvas.client.resize(sr.client, {height : "1000px", width : "900px"});
415                      * });
416                      *
417                      * @example
418                      * //Set only the height
419                      * Sfdc.canvas(function() {
420                      *     sr = JSON.parse('<%=signedRequestJson%>');
421                      *     Sfdc.canvas.client.resize(sr.client, {height : "1000px"});
422                      * });
423                      *
424                      */
425                     resize : function(client, size) {
426                         var sh, ch, sw, cw, s = {height : "", width : ""};
427 
428                         // If the size was not supplied, adjust window
429                         if ($$.isNil(size)) {
430                             sh = document.documentElement.scrollHeight;
431                             ch = document.documentElement.clientHeight;
432                             if (ch !== sh) {
433                                 s.height = sh + "px";
434                             }
435                             sw = document.documentElement.scrollWidth;
436                             cw = document.documentElement.clientWidth;
437                             if (sw !== cw) {
438                                 s.width = sw + "px";
439                             }
440                         }
441                         else {
442                             if (!$$.isNil(size.height)) {
443                                 s.height = size.height;
444                             }
445                             if (!$$.isNil(size.width)) {
446                                 s.width = size.width;
447                             }
448                         }
449                         if (!$$.isNil(s.height) || !$$.isNil(s.width)) {
450                             postit(null, {type : "resize", config : {client : client}, size : s});
451                         }
452                     },
453                     /**
454                      * @public
455                      * @name Sfdc.canvas.client#autogrow
456                      * @function
457                      * @description Starts or stops a timer which will automatically check the content size of your iframe and
458                      * adjust the frame accordingly.
459                      * Use this function when you know your content is changing size, but you're not sure when. There is delay as
460                      * the resizing is done asynchronous. Therfore, if you know when your content changes size, you should call
461                      * 'resize()' explicitly and save Browser CPU cycles.
462                      * Note: you should turn off scrolling before this call otherwise you can get a flicker.
463                      * @param {client} client object from signed request.
464                      * @param {boolean} b on or off, true default
465                      * @param {Integer} interval used to check content size, default timeout is 300ms.
466                      * @example
467                      *
468                      * // Turn on auto grow with default settings.
469                      * Sfdc.canvas(function() {
470                      *     sr = JSON.parse('<%=signedRequestJson%>');
471                      *     Sfdc.canvas.client.autogrow(sr.client);
472                      * });
473                      *
474                      * // Turn on auto grow with polling interval of 100ms (milli seconds).
475                      * Sfdc.canvas(function() {
476                      *     sr = JSON.parse('<%=signedRequestJson%>');
477                      *     Sfdc.canvas.client.autogrow(sr.client, true, 100);
478                      * });
479                      *
480                      * // Turn off auto grow.
481                      * Sfdc.canvas(function() {
482                      *     sr = JSON.parse('<%=signedRequestJson%>');
483                      *     Sfdc.canvas.client.autogrow(sr.client, false);
484                      * });
485                      */
486                     autogrow : function(client, b, interval) {
487                         var ival = ($$.isNil(interval)) ? 300 : interval;
488                         autog  = ($$.isNil(b)) ? true : b;
489                         if (autog === false) {
490                             return;
491                         }
492                         setTimeout(function () {
493                             submodules.frame.resize(client);
494                             submodules.frame.autogrow(client, autog);
495                         },ival);
496                     }
497                 };
498             }());
